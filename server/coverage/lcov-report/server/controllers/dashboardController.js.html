const User = require('../models/User');
const DesignConfig = require('../models/DesignConfig');
const CleaningLog = require('../models/CleaningLog');
const asyncHandler = require('../utils/asyncHandler');
 
// GET /api/dashboard/summary
exports.getDashboardSummary = asyncHandler(async (req, res) => {
    const user = await User.findById(req.user._id);
    const designs = await DesignConfig.find({ userId: req.user._id, status: { $ne: 'draft' } });
    const cleanings = await CleaningLog.find({ userId: req.user._id }).sort({ date: -1 }).limit(5);
 
    const activeSystem = user.solarSystems.find(s => s.status === 'active');
    const latestDesign = designs[0];
 
    // Calculate simulated energy data
    const capacity = activeSystem?.capacity || latestDesign?.panelPlacement?.totalCapacity || 5;
    const annualProduction = activeSystem?.annualProduction || latestDesign?.panelPlacement?.estimatedAnnualProduction || capacity * 1400;
    const dailyAvg = annualProduction / 365;
    const monthlyAvg = annualProduction / 12;
 
    // Financial calculations
    const monthlyBill = user.properties[0]?.currentEnergyBill || 3000;
    const electricityRate = 8; // INR per kWh
    const monthlySavings = Math.min(monthlyAvg * electricityRate, monthlyBill);
    const yearsActive = activeSystem ? ((Date.now() - new Date(activeSystem.installedDate).getTime()) / (365.25 * 24 * 3600 * 1000)) : 0;
    const lifetimeSavings = monthlySavings * Math.max(yearsActive * 12, 1);
 
    // System health
    const lastCleaning = cleanings[0];
    const daysSinceClean = lastCleaning ? Math.floor((Date.now() - new Date(lastCleaning.date).getTime()) / 86400000) : 30;
    const dustEfficiency = Math.max(85, 100 - daysSinceClean * 0.5);
 
    // Environmental
    const co2PerKwh = 0.82; // kg CO2 per kWh (India grid average)
    const co2Offset = (annualProduction * Math.max(yearsActive, 1) * co2PerKwh) / 1000; // tons
 
    res.json({
        success: true,
        data: {
            energy: {
                todayGeneration: +(dailyAvg * (0.8 + Math.random() * 0.4)).toFixed(1),
                monthlyTotal: +monthlyAvg.toFixed(0),
                yearlyTotal: +annualProduction.toFixed(0),
                capacity,
                sparklineData: Array.from({ length: 30 }, (_, i) => ({
                    day: i + 1,
                    production: +(dailyAvg * (0.6 + Math.random() * 0.8)).toFixed(1),
                })),
            },
            financial: {
                monthlySavings: +monthlySavings.toFixed(0),
                lifetimeSavings: +lifetimeSavings.toFixed(0),
                breakEvenProgress: Math.min(100, +(yearsActive / 6 * 100).toFixed(1)),
                nextPayment: user.financingDetails[0] ? {
                    amount: user.financingDetails[0].emi,
                    date: new Date(Date.now() + 15 * 86400000).toISOString(),
                } : null,
            },
            systemHealth: {
                efficiency: +dustEfficiency.toFixed(1),
                dustLevel: daysSinceClean > 20 ? 'high' : daysSinceClean > 10 ? 'medium' : 'low',
                daysSinceClean,
                nextCleaningDate: new Date(Date.now() + (30 - daysSinceClean) * 86400000).toISOString(),
                panelCount: activeSystem?.panelCount || latestDesign?.panelPlacement?.panelCount || 0,
            },
            environmental: {
                co2Offset: +co2Offset.toFixed(2),
                treesEquivalent: Math.round(co2Offset * 45),
                milesNotDriven: Math.round(co2Offset * 2481),
            },
            recentCleanings: cleanings,
        },
    });
});
 
// GET /api/dashboard/energy?range=daily|weekly|monthly|yearly
exports.getEnergyData = asyncHandler(async (req, res) => {
    const { range = 'monthly' } = req.query;
    const user = await User.findById(req.user._id);
    const capacity = user.solarSystems[0]?.capacity || 5;
    const dailyAvg = capacity * 1400 / 365;
 
    let labels = [];
    let production = [];
    let consumption = [];
 
    switch (range) {
        case 'daily': {
            for (let h = 6; h <= 18; h++) {
                labels.push(`${h}:00`);
                const hourFactor = h <= 12 ? (h - 6) / 6 : (18 - h) / 6;
                production.push(+(capacity * hourFactor * (0.7 + Math.random() * 0.6)).toFixed(2));
                consumption.push(+(0.3 + Math.random() * 1.5).toFixed(2));
            }
            break;
        }
        case 'weekly': {
            const days = ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'];
            for (const day of days) {
                labels.push(day);
                production.push(+(dailyAvg * (0.6 + Math.random() * 0.8)).toFixed(1));
                consumption.push(+(dailyAvg * 0.7 * (0.5 + Math.random() * 1)).toFixed(1));
            }
            break;
        }
        case 'yearly': {
            const months = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
            const seasonalFactor = [0.7, 0.8, 0.95, 1.1, 1.2, 1.15, 0.9, 0.85, 1.0, 0.95, 0.8, 0.7];
            for (let i = 0; i < 12; i++) {
                labels.push(months[i]);
                production.push(+(dailyAvg * 30 * seasonalFactor[i]).toFixed(0));
                consumption.push(+(dailyAvg * 30 * 0.7 * (0.8 + Math.random() * 0.4)).toFixed(0));
            }
            break;
        }
        default: { // monthly
            for (let d = 1; d <= 30; d++) {
                labels.push(`Day ${d}`);
                production.push(+(dailyAvg * (0.6 + Math.random() * 0.8)).toFixed(1));
                consumption.push(+(dailyAvg * 0.7 * (0.5 + Math.random() * 1)).toFixed(1));
            }
        }
    }
 
    res.json({
        success: true,
        data: { labels, production, consumption, range },
    });
});

exports.getSavingsData = asyncHandler(async (req, res) => {
    const user = await User.findById(req.user._id);
    const monthlyBill = user.properties[0]?.currentEnergyBill || 3000;
    const capacity = user.solarSystems[0]?.capacity || 5;
    const monthlySavings = Math.min(capacity * 1400 / 12 * 8, monthlyBill);
 
    const cumulativeSavings = [];
    const projectedSavings = [];
    let cumulative = 0;
    let projected = 0;
 
    for (let m = 1; m <= 300; m++) { // 25 years
        const degradation = 1 - 0.005 * Math.floor(m / 12);
        const actualMonthly = monthlySavings * degradation * (0.85 + Math.random() * 0.3);
        cumulative += actualMonthly;
        projected += monthlySavings * degradation;
 
        if (m % 12 === 0) {
            cumulativeSavings.push({ year: m / 12, actual: +cumulative.toFixed(0), projected: +projected.toFixed(0) });
        }
    }
 
    res.json({
        success: true,
        data: {
            monthlySavings: +monthlySavings.toFixed(0),
            cumulativeSavings,
        },
    });
});

exports.getEnvironmentalImpact = asyncHandler(async (req, res) => {
    const user = await User.findById(req.user._id);
    const capacity = user.solarSystems[0]?.capacity || 5;
    const annualProduction = capacity * 1400;
    const co2Factor = 0.82;
 
    const yearsData = [];
    for (let y = 1; y <= 25; y++) {
        const yearProd = annualProduction * (1 - 0.005 * y);
        yearsData.push({
            year: y,
            co2Offset: +(yearProd * co2Factor / 1000).toFixed(2),
            treesEquivalent: Math.round(yearProd * co2Factor / 1000 * 45),
        });
    }
 
    const totalCo2 = yearsData.reduce((acc, y) => acc + y.co2Offset, 0);
 
    res.json({
        success: true,
        data: {
            totalCo2Offset: +totalCo2.toFixed(2),
            totalTreesEquivalent: Math.round(totalCo2 * 45),
            totalMilesNotDriven: Math.round(totalCo2 * 2481),
            yearlyData: yearsData,
        },
    });
});
