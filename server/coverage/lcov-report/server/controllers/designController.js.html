const DesignConfig = require('../models/DesignConfig');
const ApiError = require('../utils/ApiError');
const asyncHandler = require('../utils/asyncHandler');
const axios = require('axios');
 
const AI_SERVICE_URL = process.env.AI_SERVICE_URL || 'http://localhost:8000';
 
// POST /api/design/roof-analysis
exports.analyzeRoof = asyncHandler(async (req, res) => {
    const { lat, lng, propertyIndex = 0 } = req.body;
 
    let analysisResult;
 
    try {
        if (req.file) {
            const formData = new FormData();
            formData.append('file', new Blob([req.file.buffer]), req.file.originalname);
            const response = await axios.post(`${AI_SERVICE_URL}/ai/roof-analysis`, formData, {
                headers: { 'Content-Type': 'multipart/form-data' },
                timeout: 30000,
            });
            analysisResult = response.data;
        } else if (lat && lng) {
            const response = await axios.post(`${AI_SERVICE_URL}/ai/roof-analysis`, {
                lat: parseFloat(lat),
                lng: parseFloat(lng),
            }, { timeout: 30000 });
            analysisResult = response.data;
        } else {
            throw ApiError.badRequest('Provide either an image file or coordinates (lat, lng)');
        }
    } catch (err) {
        if (err instanceof ApiError) throw err;
        // Fallback demo analysis
        const roofArea = req.body.roofArea || 100;
        analysisResult = {
            success: true,
            data: {
                totalArea: roofArea,
                usableArea: roofArea * 0.75,
                obstructions: [
                    { type: 'chimney', area: roofArea * 0.05, position: { x: 0.8, y: 0.2 } },
                    { type: 'vent', area: roofArea * 0.02, position: { x: 0.3, y: 0.7 } },
                    { type: 'shadow_zone', area: roofArea * 0.18, position: { x: 0.1, y: 0.9 } },
                ],
                roofPolygon: [[0, 0], [1, 0], [1, 1], [0, 1]],
                confidence: 0.87,
            },
        };
    }
 
    const design = await DesignConfig.create({
        userId: req.user._id,
        propertyIndex,
        roofAnalysis: analysisResult.data,
        status: 'analyzed',
    });
 
    res.status(201).json({
        success: true,
        data: { design, analysis: analysisResult.data },
    });
});
 
// POST /api/design/panel-placement
exports.getPanelPlacement = asyncHandler(async (req, res) => {
    const { designId, lat, lng, roofTilt, roofOrientation, panelWattage = 400 } = req.body;
 
    let design;
    if (designId) {
        design = await DesignConfig.findOne({ _id: designId, userId: req.user._id });
        if (!design) throw ApiError.notFound('Design not found');
    }
 
    let placementResult;
    try {
        const response = await axios.post(`${AI_SERVICE_URL}/ai/panel-placement`, {
            usable_area: design?.roofAnalysis?.usableArea || 75,
            lat: parseFloat(lat) || 28.6139,
            lng: parseFloat(lng) || 77.209,
            roof_tilt: parseFloat(roofTilt) || 15,
            roof_orientation: roofOrientation || 'south',
            panel_wattage: panelWattage,
        }, { timeout: 30000 });
        placementResult = response.data;
    } catch {
        // Fallback demo placement
        const usableArea = design?.roofAnalysis?.usableArea || 75;
        const panelArea = 2.0; // mÂ² per panel
        const panelCount = Math.floor(usableArea / panelArea);
        const capacity = (panelCount * panelWattage) / 1000;
        const annualProduction = capacity * 1400;
 
        const layout = [];
        const cols = Math.ceil(Math.sqrt(panelCount * 2));
        const rows = Math.ceil(panelCount / cols);
        for (let r = 0; r < rows; r++) {
            for (let c = 0; c < cols && layout.length < panelCount; c++) {
                layout.push({
                    row: r, col: c,
                    x: c * 1.1, y: r * 2.1,
                    width: 1.0, height: 2.0,
                    orientation: 'portrait',
                });
            }
        }
 
        placementResult = {
            success: true,
            data: {
                panelCount,
                panelWattage,
                totalCapacity: capacity,
                optimalTiltAngle: Math.abs(parseFloat(lat) || 28.6) * 0.9,
                optimalAzimuth: 180,
                estimatedAnnualProduction: annualProduction,
                layout,
                degradationSchedule: Array.from({ length: 25 }, (_, i) => ({
                    year: i + 1,
                    efficiency: +(100 - i * 0.5).toFixed(1),
                    production: +(annualProduction * (1 - i * 0.005)).toFixed(0),
                })),
            },
        };
    }
 
    if (design) {
        design.panelPlacement = placementResult.data;
        design.status = 'analyzed';
        await design.save();
    }
 
    res.json({
        success: true,
        data: { placement: placementResult.data, designId: design?._id },
    });
});
 
// GET /api/design
exports.getDesigns = asyncHandler(async (req, res) => {
    const designs = await DesignConfig.find({ userId: req.user._id }).sort({ updatedAt: -1 });
    res.json({ success: true, data: { designs } });
});
 
// GET /api/design/:id
exports.getDesign = asyncHandler(async (req, res) => {
    const design = await DesignConfig.findOne({ _id: req.params.id, userId: req.user._id });
    if (!design) throw ApiError.notFound('Design not found');
    res.json({ success: true, data: { design } });
});
 
// PUT /api/design/:id
exports.updateDesign = asyncHandler(async (req, res) => {
    const design = await DesignConfig.findOneAndUpdate(
        { _id: req.params.id, userId: req.user._id },
        req.body,
        { new: true, runValidators: true }
    );
    if (!design) throw ApiError.notFound('Design not found');
    res.json({ success: true, data: { design } });
});
 
// DELETE /api/design/:id
exports.deleteDesign = asyncHandler(async (req, res) => {
    const design = await DesignConfig.findOneAndDelete({ _id: req.params.id, userId: req.user._id });
    if (!design) throw ApiError.notFound('Design not found');
    res.json({ success: true, message: 'Design deleted' });
});
